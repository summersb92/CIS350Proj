8   import javax.swing.JButton;

11  import javax.swing.JPanel;
12  import javax.swing.JScrollPane;
13  import javax.swing.JTextArea;
14  import twitter4j.TwitterException;
15  
16  import engine.TwitterEngine;
17  
18  public class DirectMessageSender {
19  
20      private JFrame MessageEditor;
21  
22      private JPanel MessagePanel;
23  
24      private JTextArea MessageArea;
25  
26      private JButton SendMessageButton;
27  
28      private TwitterEngine engine;
29  
30      private String message;
31  
32      private long userId;
33  
34      public DirectMessageSender(long userId, TwitterEngine engine) {
35          this.engine = engine;
36          this.userId = userId;
37          ButtonListener listener = new ButtonListener();
38  
39          MessageEditor = new JFrame("Send Direct Message");
40  
41          MessagePanel = new JPanel();
42          MessagePanel.setLayout(new BorderLayout());
43  
44          MessageArea = new JTextArea();
45          MessageArea.setLineWrap(true);
46  
47          JScrollPane scrollpane = new JScrollPane(MessageArea);
48  
49          SendMessageButton = new JButton("Send Message");
50          SendMessageButton.addActionListener(listener);
51  
52          MessagePanel.add(scrollpane, BorderLayout.CENTER);
53          MessagePanel.add(SendMessageButton, BorderLayout.PAGE_END);
54  
55          MessageEditor.add(MessagePanel);
56  
57          MessageEditor.setSize(300, 200);
58  
59          MessageEditor.setVisible(true);
60      }
61  
62      class ButtonListener implements ActionListener {
63          public void actionPerformed(final ActionEvent e) {
64  
65              if (e.getSource().equals(SendMessageButton)) {
66                  message = MessageArea.getText();
67                  try {
68                      engine.sendDirectMessage(userId, message);
69                      MessageEditor.dispatchEvent(new WindowEvent(MessageEditor,
70                              WindowEvent.WINDOW_CLOSING));
71                  } catch (TwitterException e1) {
72                      if (e1.exceededRateLimitation()) {
73                          JOptionPane.showMessageDialog(MessageEditor,
74                                  "You message is longer than 140 characters."
75                                          + "  Please make your message shorter");
76                      }
77                      e1.printStackTrace();
78  
79                  }
80              }
81          }
82      }
