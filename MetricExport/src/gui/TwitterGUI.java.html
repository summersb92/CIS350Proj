542     class ButtonListener implements ActionListener {
543         public void actionPerformed(final ActionEvent e) {
544             if (e.getSource().equals(LoginButton)) {
545                 try {
546                     tabs.removeAll();
547                     engine.logout();
548 
549                     engine.login();
550                     profileTabInit();
551                     postTimeTabInit();
552                     followerTabInit();
553                     favoritesTabInit();
554                     MyTweetsTabInit();
555                     FriendsListTabInit();
556 
557                     tabs.addTab("Profile", ProfilePanel);
558                     tabs.addTab("Post Tweet/Timeline", PostTimePanel);
559                     tabs.addTab("Followers", TwitResults);
560                     tabs.addTab("Friends List", FriendsListPanel);
561                     tabs.addTab("Favorites", FavoritesPanel);
562                     tabs.addTab("Your Tweets", MyTweetsPanel);
563 
564                     guiFrame.add(tabs);
565 
566                 } catch (FileNotFoundException e1) {
567                     e1.printStackTrace();
568                 } catch (IllegalStateException e1) {
569                     e1.printStackTrace();
570                 } catch (TwitterException e1) {
571                     e1.printStackTrace();
572                 } catch (IOException e1) {
573                     e1.printStackTrace();
574                 } catch (Exception e1) {
575                     e1.printStackTrace();
576                 }
577             }
578             if (e.getSource().equals(SignoutButton)) {
579                 engine.logout();
580 
581                 System.exit(0);
582             }
583 
584             if (e.getSource().equals(TweetButton)) {
585                 try {
586                     String post = updateTextBox.getText();
587                     engine.postStatus(post);
588                     updateTextBox.setText("");
589 
590                 } catch (TwitterException e1) {
591                     e1.printStackTrace();
592                 }
593             }
594             if (e.getSource().equals(updateTimelineButton)) {
595                 try {
596                     tabs.remove(1);
597                     postTimeTabInit();
598 
599                     tabs.insertTab("Post Tweet/Timeline", null, PostTimePanel,
600                             null, 1);
601 
602                     tabs.setSelectedIndex(1);
603                 } catch (IllegalStateException  e1) {
604                     e1.printStackTrace();
605                 } catch (TwitterException e1) {
606                     // TODO Auto-generated catch block
607                     e1.printStackTrace();
608                 }
609             }
610 
611             if (e.getSource().equals(UpdateFavoritesButton)) {
612 
613                 try {
614 
615                     tabs.remove(4);
616                     favoritesTabInit();
617 
618                     tabs.insertTab("Favorites", null, FavoritesPanel, null, 4);
619 
620                     tabs.setSelectedIndex(4);
621 
622                 } catch (IllegalStateException e1) {
623                     e1.printStackTrace();
624                 } catch (TwitterException e1) {
625                     // TODO Auto-generated catch block
626                     e1.printStackTrace();
627                 }
628             }
629 
630             if (e.getSource().equals(MyTweetsFavoriteButton)) {
631                 try {
632                     engine.favoriteTweet(MyTweetsStatusIds[list
633                             .getSelectedIndex()]);
634                 } catch (NumberFormatException  e1) {
635                     e1.printStackTrace();
636                 } catch (TwitterException e1) {
637                     // TODO Auto-generated catch block
638                     e1.printStackTrace();
639                 }
640 
641             }
642 
643             if (e.getSource().equals(DeleteMyTweetButton)) {
644 
645                 try {
646                     engine.deleteStatus(MyTweetsStatusIds[list
647                             .getSelectedIndex()]);
648                 } catch (TwitterException e1) {
649                     e1.printStackTrace();
650                 }
651 
652             }
653 
654             if (e.getSource().equals(VeiwProfile)) {
655                 try {
656                     new FriendVeiwer(userIds[FriendsList.getSelectedIndex()],
657                             engine);
658                 } catch (TwitterException e1) {
659                     e1.printStackTrace();
660                 } catch (Exception e1) {
661                     e1.printStackTrace();
662                 }
663             }
664 
665             if (e.getSource().equals(RemoveFriend)) {
666                 try {
667                     engine.removefriend(userIds[FriendsList.getSelectedIndex()]);
668                 } catch (TwitterException e1) {
669                     e1.printStackTrace();
670                 }
671             }
672 
673             if (e.getSource().equals(ViewMessagesButton)) {
674                 try {
675                     new DirectMessageViewer(engine);
676                 } catch (TwitterException e1) {
677                     e1.printStackTrace();
678                 }
679             }
680 
681             if (e.getSource().equals(TimelineFavoriteButton)) {
682                 try {
683                     engine.favoriteTweet(MyTimeLineStatusIds[TimelineList
684                             .getSelectedIndex()]);
685                 } catch (NumberFormatException e1) {
686                     e1.printStackTrace();
687                 } catch (TwitterException e1) {
688                     if (e1.getErrorCode() == 139) {
689                         JOptionPane.showMessageDialog(guiFrame,
690                                 "You have already favorited that Tweet",
691                                 "Error", JOptionPane.ERROR_MESSAGE);
692                     }
693                 }
694             }
695 
696             if (e.getSource().equals(DeleteFavoriteButton)) {
697                 try {
698                     engine.deleteStatus(MyFavoriteStatusIds[FavoritesList
699                             .getSelectedIndex()]);
700                 } catch (TwitterException e1) {
701                     e1.printStackTrace();
702                 }
703             }
704 
705             if (e.getSource().equals(UpdateFriendsList)) {
706                 tabs.remove(3);
707                 try {
708                     FriendsListTabInit();
709                 } catch (TwitterException e1) {
710 
711                     e1.printStackTrace();
712                 }
713 
714                 tabs.insertTab("Friends", null, FriendsListPanel, null, 3);
715 
716                 tabs.setSelectedIndex(3);
717             }
718 
719             if (e.getSource().equals(UpdateMyTweetsList)) {
720                 tabs.remove(5);
721                 try {
722                     favoritesTabInit();
723                 } catch (TwitterException e1) {
724                     e1.printStackTrace();
725                 }
726 
727                 tabs.insertTab("My Tweets", null, MyTweetsPanel, null, 5);
728 
729                 tabs.setSelectedIndex(5);
730             }
731         }
732     }
