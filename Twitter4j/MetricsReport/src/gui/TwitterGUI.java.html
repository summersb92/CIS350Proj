164     /**
165      * Handels all of the actions in the JMenuBar
166      */
167     private ActionListener menuHandeler = new 
168             ActionListener(){
169         @Override
170         public void actionPerformed(ActionEvent e) {
171             //System.out.println(e);
172             if(e.getActionCommand().equals
173                     ("Delete Status")){
174                 engine.deleteStatus();
175             }
176             if(e.getActionCommand().equals
177                     ("Delete Table Status")){
178                 engine.deleteTweet();
179             }
180             if(e.getActionCommand().equals
181                     ("Export to XML ...")){
182                 saveButtonAction();
183             }
184             if(e.getActionCommand().equals("Quit")){
185                 System.exit(0);
186             }
187             if(e.getActionCommand().equals("About...")){
188                 JOptionPane.showMessageDialog(null, 
189                         "Produced by Benjamin Summers \n" +
190                         "                  10/28/2011 \n" +
191                         "         For a JTwitterProgram");
192             }
193         }   
194     };
